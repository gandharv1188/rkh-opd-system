## 5. Public API
```ts
export interface DisLogContext {
  correlation_id?: string; extraction_id?: string; operator_id?: string; stage?: string;
}
export interface CreateLoggerOptions {
  level?: LoggerOptions['level'];
  redactPaths?: readonly string[];
  destination?: pino.DestinationStream; // test hook
}
export function createLogger(opts?: CreateLoggerOptions): Logger;
export function withContext(base: Logger, ctx: DisLogContext): Logger;
```
